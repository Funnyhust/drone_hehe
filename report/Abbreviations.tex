\section*{DANH MỤC KÝ HIỆU VÀ CHỮ VIẾT TẮT}
\addcontentsline{toc}{section}{\numberline{}DANH MỤC KÝ HIỆU VÀ CHỮ VIẾT TẮT}

\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{longtable}{|l|p{11.5cm}|}
\hline
\textbf{Ký hiệu} & \textbf{Ý nghĩa đầy đủ} \\ \hline
\endhead
ADC & Analog-to-Digital Converter - Bộ chuyển đổi tương tự sang số \\ \hline
AHRS & Attitude and Heading Reference System - Hệ thống tham chiếu tư thế và hướng bay \\ \hline
AI & Artificial Intelligence - Trí tuệ nhân tạo \\ \hline
BEC & Battery Elimination Circuit - Mạch hạ áp xung cấp nguồn logic tích hợp trên ESC hoặc FC \\ \hline
CSV & Comma-Separated Values - Định dạng tệp dữ liệu phân tách bằng dấu phẩy \\ \hline
DMA & Direct Memory Access - Cơ chế truy cập bộ nhớ trực tiếp \\ \hline
EEPROM & Electrically Erasable Programmable Read-Only Memory - Bộ nhớ chỉ đọc có thể xóa và lập trình bằng điện \\ \hline
ELRS & ExpressLRS - Hệ thống điều khiển vô tuyến tầm xa độ trễ thấp \\ \hline
EMI & Electromagnetic Interference - Nhiễu điện từ trường phát sinh từ động cơ/ESC \\ \hline
ESC & Electronic Speed Controller - Bộ điều tốc động cơ điện tử \\ \hline
FC & Flight Controller - Bo mạch điều khiển bay trung tâm \\ \hline
FPS & Frames Per Second - Số khung hình xử lý trên một giây \\ \hline
FPV & First Person View - Góc nhìn thứ nhất (truyền hình ảnh trực tiếp từ camera) \\ \hline
FSM & Finite State Machine - Thiết kế máy trạng thái hữu hạn \\ \hline
GCS & Ground Control Station - Trạm điều khiển và điều phối mặt đất \\ \hline
GPIO & General Purpose Input/Output - Cổng vào/ra đa dụng của vi điều khiển \\ \hline
GPS & Global Positioning System - Hệ thống định vị toàn cầu \\ \hline
HTML & HyperText Markup Language - Ngôn ngữ đánh dấu siêu văn bản (xuất báo cáo cứu hộ) \\ \hline
I2C & Inter-Integrated Circuit - Giao tiếp nối tiếp đồng bộ giữa các vi mạch \\ \hline
IIR & Infinite Impulse Response - Bộ lọc số có đáp ứng xung vô hạn (lọc khâu vi phân D) \\ \hline
IMU & Inertial Measurement Unit - Đơn vị đo lường quán tính (cảm biến gia tốc và con quay hồi chuyển) \\ \hline
IoU & Intersection over Union - Chỉ số đánh giá độ tương đồng vùng bao (sử dụng trong ByteTrack) \\ \hline
LDO & Low Dropout Regulator - Bộ ổn áp tuyến tính hiệu số cực thấp \\ \hline
LED & Light Emitting Diode - Đi-ốt phát quang báo trạng thái lỗi \\ \hline
MCU & Microcontroller Unit - Vi điều khiển xử lý trung tâm \\ \hline
OSD & On-Screen Display - Bộ hiển thị ký tự lớp phủ trên màn hình video \\ \hline
PCB & Printed Circuit Board - Bảng mạch in điện tử \\ \hline
PID & Proportional-Integral-Derivative - Thuật toán điều khiển Tỉ lệ - Tích phân - Vi phân \\ \hline
PWM & Pulse Width Modulation - Phương pháp điều chế độ rộng xung \\ \hline
ROI & Region of Interest - Vùng quan tâm đặc biệt được thiết lập trên camera \\ \hline
RTOS & Real-Time Operating System - Hệ điều hành thời gian thực \\ \hline
SCL & Serial Clock - Đường truyền xung nhịp của giao tiếp bus hai dây I2C \\ \hline
SDA & Serial Data - Đường truyền dữ liệu của giao tiếp bus hai dây I2C \\ \hline
SPI & Serial Peripheral Interface - Giao tiếp ngoại vi nối tiếp tốc độ cao \\ \hline
UAV & Unmanned Aerial Vehicle - Thiết bị bay không người lái \\ \hline
UART & Universal Asynchronous Receiver Transmitter - Bộ truyền nhận nối tiếp không đồng bộ \\ \hline
VRX & Video Receiver - Bộ nhận tín hiệu video FPV không dây tại trạm mặt đất \\ \hline
VTX & Video Transmitter - Bộ phát tín hiệu video FPV không dây trên Drone \\ \hline
YOLO & You Only Look Once - Thuật toán phát hiện đối tượng thời gian thực \\ \hline
\end{longtable}
\end{center}
